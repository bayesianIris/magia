create_magica_spiritual_head_effect = {
	# Create a new unlanded cleric as the Head of Faith
	create_character = {
		location = root.capital_province
		age = { 30 50 }
		gender_female_chance = root_faith_clergy_gender_female_chance
		trait = education_learning_4
		faith = root.faith
		culture = root.culture
		learning = { 14 22 }
		save_scope_as = new_magica_head
	}

	# Create or return the fixed HoF title and grant it
	if = {
		limit = { title:k_magica_holy_see = { is_title_created = yes } }
		create_title_and_vassal_change = {
			type = returned
			save_scope_as = change
			add_claim_on_loss = no
		}
	}
	else = {
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
	}
	title:k_magica_holy_see = {
		change_title_holder = {
			holder = scope:new_magica_head
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change

	# Convert all Magica faiths to spiritual head and set the shared HoF title
	religion:magica_religion = {
		every_faith = {
			if = {
				limit = { has_doctrine = doctrine_no_head }
				remove_doctrine = doctrine_no_head
			}
			if = {
				limit = { has_doctrine = doctrine_temporal_head }
				remove_doctrine = doctrine_temporal_head
			}
			if = {
				limit = { NOT = { has_doctrine = doctrine_spiritual_head } }
				add_doctrine = doctrine_spiritual_head
			}
			set_religious_head_title = title:k_magica_holy_see
		}
	}
}
