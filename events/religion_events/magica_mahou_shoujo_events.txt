namespace = magica_mahou_shoujo_selection

# 魔法少女候选人选择事件
# 在每个信仰 magica_religion 的 12-16 岁小孩身上触发
# 用于随机决定谁成为已被证实的魔法少女

magica_mahou_shoujo_selection.0001 = {
	type = character_event
	title = magica_mahou_shoujo_selection.0001.t
	desc = magica_mahou_shoujo_selection.0001.desc
	theme = faith
	
	left_portrait = {
		character = root
		animation = personality_honorable
	}

	trigger = {
		age >= 10
		age <= 16
		is_female = yes
		faith.religion = religion:magica_religion
		NOT = { has_trait = magica_mahou_shoujo }
	}

	immediate = {
		save_scope_as = candidate_girl
	}

	option = {
		name = magica_mahou_shoujo_selection.0001.a
		add_trait = magica_mahou_shoujo
		add_trait = immortal
		if = {
			limit = {
				NOT = { has_game_rule = magica_mahou_shoujo_rules_disabled }
			}
			add_trait = immortal
		}
		add_character_modifier = {
			modifier = magica_mahou_shoujo_awakening_modifier
			years = 10
		}
		every_liege_or_above = {
			every_held_title = {
				save_scope_as = claimed_title
				root = { add_pressed_claim = scope:claimed_title }
			}
		}

		# 根据规则决定是否添加修正 trait（用于禁用模式）
		if = {
			limit = { has_game_rule = magica_mahou_shoujo_rules_disabled }
			root = { 
				add_trait = magica_mahou_shoujo_rules_correction
			}
		}
		else = {
			root = { remove_trait = magica_mahou_shoujo_rules_correction }
		}
		
		# 如果候选人出身卑微，通知其当地领主触发招募事件
		if = {
			limit = {
				NOT = { is_landed = yes }
				exists = liege
				liege = {
					is_playable_character = yes
				}
			}
			# liege = {
			# 	trigger_event = {
			# 		id = magica_mahou_shoujo_selection.0002
			# 		days = 3
			# 	}
			# 	save_scope_as = local_liege
			# }
		}
	}
	
	after = {
		if = {
			limit = { has_game_rule = magica_mahou_shoujo_rules_enabled }
			# 启动魔法少女年度压力检查循环
			magica_mahou_shoujo_stress_event_aftereffects = yes
			# 启动魔法少女随机生死决斗循环
			magica_mahou_shoujo_duel_loop_aftereffects = yes
		}
	}
}


magica_mahou_shoujo_selection.0005 = {
	type = character_event
	title = magica_mahou_shoujo_selection.0005.t
	desc = magica_mahou_shoujo_selection.0005.desc
	theme = faith
	
	left_portrait = {
		character = root
		animation = personality_curious
	}

	trigger = {
		is_ai = no
		faith.religion = religion:magica_religion
	}

	immediate = {
		save_scope_as = player_character
	}

	# 我是魔法少女
	option = {
		name = magica_mahou_shoujo_selection.0005.a

		trigger = {
			age >= 10
			age <= 16
			is_female = yes
		}
		show_as_unavailable = {
			OR = {
				age < 10
				age > 16
				is_female = no		
			}

		}
		trigger_event = magica_mahou_shoujo_selection.0001
		
	}

	# 我只是一个普通人
	option = {
		name = magica_mahou_shoujo_selection.0005.b
	}
}


# 领主招募卑微出身魔法少女的事件
magica_mahou_shoujo_selection.0002 = {
	type = character_event
	title = magica_mahou_shoujo_selection.0002.t
	desc = magica_mahou_shoujo_selection.0002.desc
	theme = court
	
	left_portrait = {
		character = root
		animation = personality_shrewd
	}

	trigger = {
		is_playable_character = yes
	}

	immediate = {
		# 获取最近获得魔法少女特质的女性
		root.realm = {
			random_realm_character = {
				limit = {
					is_female = yes
					age >= 10
					age <= 16
					has_trait = magica_mahou_shoujo
					NOT = { is_landed = yes }
					faith.religion = religion:magica_religion
					liege = root
				}
				save_scope_as = mahou_shoujo_recruit
			}
		}
	}

	option = {
		name = magica_mahou_shoujo_selection.0002.a
		# 招募为廷臣
		if = {
			limit = {
				exists = scope:mahou_shoujo_recruit
			}
			scope:mahou_shoujo_recruit = {
				add_courtier = root
				add_opinion = {
					modifier = honored_opinion
					opinion = 30
					target = root
				}
			}
		}
		add_prestige = moderate_prestige_gain
	}

	option = {
		name = magica_mahou_shoujo_selection.0002.b
		# 拒绝招募
		if = {
			limit = {
				exists = scope:mahou_shoujo_recruit
			}
			scope:mahou_shoujo_recruit = {
				add_opinion = {
					modifier = disappointed_opinion
					opinion = -10
					target = root
				}
			}
		}
	}
}
